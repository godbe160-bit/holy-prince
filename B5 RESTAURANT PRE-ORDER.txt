<script>
let users = JSON.parse(localStorage.getItem('ga_users')) || [];
let orders = JSON.parse(localStorage.getItem('ga_orders')) || [];
let cart = [];
let currentUser = null;
let timer;

const menu = [
    {id:1, name:"Godbe Burger", price:2000}, {id:2, name:"Chicken Pizza", price:3500},
    {id:3, name:"Beef Pasta", price:2500}, {id:4, name:"Suya Fries", price:1200},
    {id:5, name:"Cold Soda", price:600}
];

function showPage(id) {
    document.querySelectorAll('.container').forEach(c => c.classList.add('hidden'));
    document.getElementById(id).classList.remove('hidden');
}

function register() {
    let u = document.getElementById('regUser').value, p = document.getElementById('regPass').value, r = document.getElementById('role').value;
    if(!u || !p) return alert("Missing info");
    users.push({username:u, password:p, role:r, pic:''});
    localStorage.setItem('ga_users', JSON.stringify(users));
    alert("Success!"); showPage('loginPage');
}

function login() {
    let u = document.getElementById('loginUser').value, p = document.getElementById('loginPass').value;
    let user = users.find(x => x.username === u && x.password === p);
    if(!user) return alert("Invalid Login");
    currentUser = user;
    if(user.role === 'admin') { showPage('adminPage'); updateAdmin(); }
    else { showPage('orderPage'); renderMenu(); updateProfile(); renderCustOrders(); }
}

function updateProfile() {
    document.getElementById('profilePic').src = currentUser.pic || 'https://via.placeholder.com/100';
    document.getElementById('pNameDisplay').innerHTML = `<strong>${currentUser.username}</strong>`;
}

function uploadProfilePic(e) {
    let r = new FileReader();
    r.onload = function() {
        currentUser.pic = r.result;
        let idx = users.findIndex(u => u.username === currentUser.username);
        users[idx].pic = r.result;
        localStorage.setItem('ga_users', JSON.stringify(users));
        updateProfile();
    };
    r.readAsDataURL(e.target.files[0]);
}

function renderMenu() {
    document.getElementById('menuDisplay').innerHTML = menu.map(m => `
        <div class="menu-item">
            <span>${m.name}<br>₦${m.price}</span>
            <button class="btn-small" onclick="addToCart(${m.id})">Add</button>
        </div>`).join('');
}

function addToCart(id) {
    cart.push(menu.find(m => m.id === id));
    updateCartUI();
}

function updateCartUI() {
    document.getElementById('cartList').innerHTML = cart.map(c => `<div class="cart-item"><span>${c.name}</span><span>₦${c.price}</span></div>`).join('');
    document.getElementById('totalPrice').innerText = `Total: ₦${cart.reduce((s, i) => s + i.price, 0)}`;
}

// TOGGLES FOR LOCATION & PAYMENT
function toggleLocationFields() {
    document.getElementById('locationFields').classList.toggle('hidden', document.getElementById('deliveryType').value === 'Pickup');
}

function togglePaymentFields() {
    let method = document.getElementById('pMethod').value;
    document.getElementById('cardFields').classList.toggle('hidden', method !== 'Card');
    document.getElementById('bankBox').classList.toggle('hidden', method !== 'Transfer');
    
    if(method === 'Transfer') {
        let s = 30; clearInterval(timer);
        timer = setInterval(() => {
            s--; document.getElementById('countdown').innerText = s + "s remaining";
            if(s <= 0) { clearInterval(timer); logout(); alert("Transfer window expired!"); }
        }, 1000);
    } else {
        clearInterval(timer);
    }
}

function placeOrder() {
    if(cart.length === 0) return alert("Cart is empty");
    let delType = document.getElementById('deliveryType').value;
    let o = {
        id: Date.now(), customer: currentUser.username, items: [...cart],
        total: cart.reduce((s, i) => s + i.price, 0), time: document.getElementById('pickupTime').value,
        delivery: delType,
        address: delType === 'Door Delivery' ? document.getElementById('streetAddress').value + ", " + document.getElementById('deliveryArea').value : "Pickup",
        payment: document.getElementById('pMethod').value,
        status: 'Pending', feedback: null, confirmed: false
    };
    orders.push(o);
    localStorage.setItem('ga_orders', JSON.stringify(orders));
    cart = []; updateCartUI(); renderCustOrders(); alert("Order Placed Successfully!");
}

// FEEDBACK LOGIC
function confirmDelivery(id) {
    orders.find(o => o.id === id).confirmed = true;
    localStorage.setItem('ga_orders', JSON.stringify(orders));
    renderCustOrders();
}

function submitFeedback(id) {
    let r = document.getElementById(`rate-${id}`).value;
    let c = document.getElementById(`comm-${id}`).value;
    orders.find(o => o.id === id).feedback = { rating: r, comment: c };
    localStorage.setItem('ga_orders', JSON.stringify(orders));
    renderCustOrders();
}

function renderCustOrders() {
    let my = orders.filter(o => o.customer === currentUser.username);
    document.getElementById('custOrders').innerHTML = my.reverse().map(o => {
        let act = "";
        if(o.status === 'Completed' && !o.confirmed) act = `<button class="btn-small" onclick="confirmDelivery(${o.id})">Confirm Delivery</button>`;
        else if(o.confirmed && !o.feedback) act = `<div class="feedback-box">Rate Food: <select id="rate-${o.id}"><option>⭐⭐⭐⭐⭐</option><option>⭐⭐⭐⭐</option><option>⭐⭐⭐</option></select><input id="comm-${o.id}" placeholder="Your comment..."><button class="btn-small" onclick="submitFeedback(${o.id})">Send Feedback</button></div>`;
        else if(o.feedback) act = `<div class="feedback-box">Sent: ${o.feedback.rating}</div>`;
        return `<li><span class="badge ${o.status}">${o.status}</span><br>Order #${o.id.toString().slice(-4)} | ₦${o.total}<br>${act}</li>`;
    }).join('');
}

function updateAdmin() {
    document.getElementById('sRev').innerText = "₦" + orders.reduce((s, o) => s + o.total, 0);
    document.getElementById('sSat').innerText = orders.filter(o => o.feedback).length;

    document.getElementById('adminOrderList').innerHTML = orders.reverse().map(o => {
        let confirmTxt = o.confirmed ? `<span class="delivered-confirm">✔ DELIVERED (CONFIRMED BY CLIENT)</span>` : "";
        let feedbackTxt = o.feedback ? `<div style="color:var(--primary); font-size:12px; margin-top:5px;"><b>Feedback:</b> ${o.feedback.rating} - ${o.feedback.comment}</div>` : "";
        return `<li>
            <div style="display:flex; justify-content:space-between"><strong>${o.customer}</strong> ${confirmTxt}</div>
            <div style="font-size:13px; color:#666"><b>Items:</b> ${o.items.map(i => i.name).join(', ')}</div>
            <div style="font-size:13px; color:#666"><b>Location:</b> ${o.address}</div>
            <div style="font-size:13px; color:#666"><b>Payment:</b> ${o.payment}</div>
            ${feedbackTxt}
            <select onchange="updateStat(${o.id}, this.value)" style="width:auto; margin-top:8px; font-size:12px;">
                <option ${o.status=='Pending'?'selected':''}>Pending</option>
                <option ${o.status=='Cooking'?'selected':''}>Cooking</option>
                <option ${o.status=='Completed'?'selected':''}>Completed</option>
            </select>
        </li>`;
    }).join('');
}

function updateStat(id, s) {
    orders.find(o => o.id === id).status = s;
    localStorage.setItem('ga_orders', JSON.stringify(orders));
    updateAdmin();
}

function logout() { location.reload(); }
</script>
